### 2.7 普通BOM模式 `synInputAndOutOnStepByBOM`

```java
if (!this.enabledCheckInputMaterial()) {
    return
}
var semiProductIds = BillOfMaterial.querySemiProductIdsByBom(context())
Group<InputOnStep>().forEach{ it.isDirty = true }
Group<OutputOnStep>().forEach{ it.isDirty = true }
var toAddInput = new List<InputOnStep>()
var toAddOutput = new List<OutputOnStep>()
Group<WorkOrder>().groupBy{ it.productId + it.bomVer }.forEach((productIdAbomver,orders) -> {
    var firstOrder = orders.firstOrNull()
    if (firstOrder == null) {
        return
    }
    var productId = firstOrder.productId
    var ktRows = KT<KTInputConstraint>().select(it.productId == productId,it.stepSequenceNr != "")
    //1.0对有特殊约束的KT，在对应步骤创建input
    ktRows.forEach{ ktRow ->
        var boms = BillOfMaterial.queryByProductIdAndRawMaterialId(context(),productId,ktRow.rawMaterialId).filter{ it.isEffective }
        if (boms.isEmpty()) {
            return
        }
        indexing(WorkOrderStep.indexByProductIdAndSeqNr,productId,ktRow.stepSequenceNr).forEach{ workOrderStep ->
            if (workOrderStep == null) {
                return
            }
            var order = workOrderStep.owner
            if (order == null || !this.canUseBomByInputProductId(order,semiProductIds,ktRow.rawMaterialId)) {
                return
            }
            boms.forEach{ bom ->
                var input = indexing(InputOnStep.indexByOwnerAndMaterialIdAndLineNr,workOrderStep,bom.rawMaterialId,bom.lineNr).firstOrNull()
                if (input == null) {
                    input = new InputOnStep(workOrderStep)
                    toAddInput.add(input)
                }
                input.isDirty = false
                input.updateAttributeByBom(Utils.dataValuesToMap(bom))
            }
        }
    }
    //2.0根据产品代码查找BOM，并对普通BOM在第一步创建input
    var bomsData = BillOfMaterial.queryByProductId(context(),productId).filter{ it.isEffective }
    var hasConstraintRawMaterialIds = ktRows.map{ it.rawMaterialId }
    var commonBoms = bomsData.filter{ !hasConstraintRawMaterialIds.contain(it.rawMaterialId) }
    orders.forEach{ order ->
        indexing(Piece.indexByOwner,order).forEach{ piece ->
            var workOrderStep = piece.firstWorkOrderStep
            if (workOrderStep == null) {
                return
            }
            commonBoms.forEach{ bom ->
                if (!this.canUseBomByInputProductId(order,semiProductIds,bom.rawMaterialId)) {
                    return
                }
                var input = indexing(InputOnStep.indexByOwnerAndMaterialIdAndLineNr,workOrderStep,bom.rawMaterialId,bom.lineNr).firstOrNull()
                if (input == null) {
                    input = toAddInput.find{ it.owner == workOrderStep && it.materialId == bom.rawMaterialId && it.lineNr == bom.lineNr }
                }
                if (input == null) {
                    input = new InputOnStep(workOrderStep)
                    toAddInput.add(input)
                }
                input.isDirty = false
                input.updateAttributeByBom(Utils.dataValuesToMap(bom))
            }
        }
    }
    //3.0对于所有订单的所有末端步骤，创建output
    orders.forEach{ order ->
        var orderProductId = order.productId
        indexing(Piece.indexByOwner,order).forEach{ piece ->
            var workOrderStep = piece.selectedWorkOrderSteps.sortedBy{ it.sequenceNr }.lastOrNull()
            if (workOrderStep == null) {
                return
            }
            var output = indexing(OutputOnStep.indexByOwnerAndProductId,workOrderStep,orderProductId).firstOrNull()
            if (output == null) {
                output = toAddOutput.find{ it.owner == workOrderStep && Utils.anyToString(it.productId) == Utils.anyToString(orderProductId) }
            }
            if (output == null) {
                output = new OutputOnStep(workOrderStep)
                toAddOutput.add(output)
            }
            output.isDirty = false
            if (bomsData.isNotEmpty()) {
                output.updateAttributeByBom(Utils.dataValuesToMap(bomsData.firstOrNull()))
            }
        }
    }
})
Group<InputOnStep>().removeIf{ it.isDirty }
Group<OutputOnStep>().removeIf{ it.isDirty }
Group<InputOnStep>().addAll(toAddInput)
Group<OutputOnStep>().addAll(toAddOutput)
```

### 2.10 库存转供应 `synInventoryToMaterialSupply`

```java
var inventories = InventoryIn.queryAllData(context())
var toAdd = new List<MaterialSupply>()
var dh = context().getDbHelper()
var semiProductIds = BillOfMaterial.querySemiProductIdsByBom(context())
inventories.groupBy{ it.productId }.forEach((productId,inventoryList) -> {
    var product = indexing(Product.indexByProductId,productId).firstOrNull()
    var detectDuration = product?.detectDuration ?? Duration.ZERO
    var canSelfProduction = product?.canSelfProduction ?? false
    var isBanChengPing = semiProductIds.contain(productId)
    inventoryList.forEach{ inventory ->
        if (isBanChengPing) {
            var semiQty = inventory.quantity
            if (semiQty > Constant.EPS) {
                var obj = new MaterialSupply()
                MaterialSupply.updateAttributeBySemiInventoryNode(obj,inventory)
                obj.detectDuration = detectDuration
                obj.canSelfProduction = canSelfProduction
                this.setSemiFinishedInventoryDate(obj,inventory)
                this.setSemiProductVirtualQuantity(obj,productId)
                MaterialSupply.updateSupplySortKey(obj)
                if (obj.supplyQuantity > Constant.EPS) {
                    toAdd.add(obj)
                }
            }
            return
        }
        var availableQty = inventory.quantity
        if (availableQty > Constant.EPS) {
            var obj = new MaterialSupply()
            MaterialSupply.updateAttributeByInventoryAvailableNode(obj,inventory)
            obj.detectDuration = detectDuration
            obj.canSelfProduction = canSelfProduction
            obj.availableReleaseTime = obj.availableTime
            obj.availableReleaseDate = obj.availableDate
            MaterialSupply.updateSupplySortKey(obj)
            if (obj.supplyQuantity > Constant.EPS) {
                toAdd.add(obj)
            }
        }
        var detectQty = inventory.detectQuantity
        if (detectQty > Constant.EPS) {
            var obj = new MaterialSupply()
            MaterialSupply.updateAttributeByInventoryDetectNode(obj,inventory)
            obj.detectDuration = detectDuration
            obj.canSelfProduction = canSelfProduction
            obj.availableReleaseTime = TimeCalc.timePlusDuration(obj.availableTime,detectDuration)
            obj.availableReleaseDate = obj.availableReleaseTime.toDate()
            MaterialSupply.updateSupplySortKey(obj)
            if (obj.supplyQuantity > Constant.EPS) {
                toAdd.add(obj)
            }
        }
    }
})
dh.insertBatch(toAdd)
```

### 2.19 冻结分配重占 `reOccupyFrozenMaterialFulfill`

```java
var dh = context().getDbHelper()
var needUpdateDemands = new List<MaterialDemand>()
var needUpdateSupplies = new List<MaterialSupply>()
var invalidFrozenFulfills = new List<MaterialFulfill>()
var qfulfill = QMaterialFulfill.qmaterialfulfill
var frozenFulfills = dh.dslSelectFrom(qfulfill).where(qfulfill.frozenFlag.eq(true)).findAll()
var supplies = MaterialSupply.queryAllData(context())
var supplyMap = supplies.groupBy{ it.uniqueId }
frozenFulfills.groupBy{ it.demandUniqueId }.forEach((demandUniqueId,fulfills) -> {
    var demand = MaterialDemand.findByUniqueId(context(),demandUniqueId)
    if (demand == null) {
        invalidFrozenFulfills.addAll(fulfills)
        return
    }
    var firstPriorityDemand = MaterialDemand.findFirstReplacePriorityByMaterialDemand(context(),demand)
    var noLocationIds = MaterialSupply.queryNoLocationIdsByProductId(context(),demand.productId,demand.pieceSizeQuantity,demand.routingId)
    var isGuojiBanchengpingDemand = MaterialSupply.isGuojiBanchengpingMaterialDemand(demand,context())
    var handBooks = new List<KTProcessingTradeHandBook>()
    var candidateMaterialIds = new List<String>()
    if (isGuojiBanchengpingDemand) {
        handBooks = MaterialSupply.queryProcessingTradeHandBooks(context(),demand.productId,demand.materialId)
        candidateMaterialIds = handBooks.map{ it.materialId ?? "" }.filter{ it != "" }
        var demandMaterialId = demand.materialId ?? ""
        if (!candidateMaterialIds.contain(demandMaterialId)) {
            candidateMaterialIds.add(demandMaterialId)
        }
    }
    var hasValidFulfill = false
    fulfills.forEach{ fulfill ->
        var supply = supplyMap.getOrDefault(fulfill.supplyUniqueId,new List<MaterialSupply>()).firstOrNull()
        if (supply == null) {
            invalidFrozenFulfills.add(fulfill)
            return
        }
        //24039：校验冻结库存供应是否命中禁用库位
        var locationId = supply.sapStockingPointId ?? ""
        var invalidLocation = MaterialSupply.isInventorySupply(supply) && locationId != "" && noLocationIds.contain(locationId)
        if (invalidLocation) {
            invalidFrozenFulfills.add(fulfill)
            return
        }
        //24040：国际订单半成品校验手册物料和库位
        if (isGuojiBanchengpingDemand && !handBooks.isEmpty() && MaterialSupply.isInventorySupply(supply)) {
            var handBookSupplies = supplies.filter{ candidateMaterialIds.contain(it.productId ?? "") }
            var legalHandBookSupplies = new List<MaterialSupply>()
            handBookSupplies.forEach{ handBookSupply ->
                var handBookLocationId = handBookSupply.sapStockingPointId ?? ""
                var handBookInvalidLocation = MaterialSupply.isInventorySupply(handBookSupply) && handBookLocationId != "" && noLocationIds.contain(handBookLocationId)
                if (!handBookInvalidLocation) {
                    legalHandBookSupplies.add(handBookSupply)
                }
            }
            MaterialSupply.filterProcessingTradeHandBookSuppliesInPlace(legalHandBookSupplies,handBooks)
            if (!legalHandBookSupplies.contain(supply)) {
                invalidFrozenFulfills.add(fulfill)
                return
            }
        }
        demand.fulfillQuantity += fulfill.quantity
        hasValidFulfill = true
        if (firstPriorityDemand != null) {
            var demandQty = demand.pieceStepQuantity * demand.inputFactor
            var ratio = demandQty > 0.0 ? fulfill.quantity / demandQty : 0.0
            firstPriorityDemand.leftRatio -= ratio
        }
        supply.leftQuantity -= fulfill.quantity
        supply.leftQuantity.roundHalfDown(7)
        if (!needUpdateSupplies.contain(supply)) {
            needUpdateSupplies.add(supply)
        }
        fulfill.availableDate = supply.availableReleaseDate
        fulfill.availableTime = supply.availableReleaseTime
    }
    if (hasValidFulfill) {
        if (!needUpdateDemands.contain(demand)) {
            needUpdateDemands.add(demand)
        }
        if (firstPriorityDemand != null && firstPriorityDemand != demand && !needUpdateDemands.contain(firstPriorityDemand)) {
            needUpdateDemands.add(firstPriorityDemand)
        }
    }
})
if (!invalidFrozenFulfills.isEmpty()) {
    dh.deleteBatch(invalidFrozenFulfills)
}
var validFrozenFulfills = frozenFulfills.filter{ !invalidFrozenFulfills.contain(it) }
if (!needUpdateDemands.isEmpty()) {
    dh.updateBatch(needUpdateDemands)
}
if (!needUpdateSupplies.isEmpty()) {
    dh.updateBatch(needUpdateSupplies)
}
if (!validFrozenFulfills.isEmpty()) {
    dh.updateBatch(validFrozenFulfills)
}
```
